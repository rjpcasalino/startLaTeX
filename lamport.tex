\section*{AI explains \LaTeX{} Length Units}

\begin{itemize}

  \item[\texttt{cm}] \textbf{Centimeters (physical world friendly)}\\
  \rule{3cm}{1pt}\\
  ``About the width of a couple of fingers.''  
  Great for print layouts where rulers still matter.

  \item[\texttt{mm}] \textbf{Millimeters (precision nerd mode)}\\
  \rule{10mm}{1pt}\\
  ``For when centimeters feel too large.''

  \item[\texttt{in}] \textbf{Inches (imperial nostalgia)}\\
  \rule{1in}{1pt}\\
  ``One inch.''

  \item[\texttt{pt}] \textbf{Points (typographic atoms)}\\
  \rule{72pt}{1pt}\\
  ``72 points = 1 inch.''

  \item[\texttt{pc}] \textbf{Picas (printer wizardry)}\\
  \rule{6pc}{1pt}\\
  ``6 picas = 1 inch.''

  \item[\texttt{em}] \textbf{Em (font-relative chameleon)}\\
  \rule{5em}{1pt}\\
  ``As wide as the letter ‘M’ — roughly.''

  \item[\texttt{ex}] \textbf{Ex (x-height mysticism)}\\
  \rule{10ex}{1pt}\\
  ``Based on the height of the letter ‘x’.''

  \item[\texttt{bp}] \textbf{Big Points (PostScript cousin)}\\
  \rule{72bp}{1pt}\\
  ``Almost like points, but not quite.
  Exists to keep standards committees employed.''

  \item[\texttt{dd}] \textbf{Didôt Points (European elegance)}\\
  \rule{10dd}{1pt}\\
  ``Metric-flavored typography.''

  \item[\texttt{sp}] \textbf{Scaled Points (TeX’s quantum unit)}\\
  \rule{65536sp}{1pt}\\
  ``1 pt = 65,536 sp.
  You will never use this intentionally.''

\end{itemize}
